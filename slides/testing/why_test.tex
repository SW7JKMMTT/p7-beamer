\section{Testing}
\author{Mathias Sass Michno}
\subsection{Why We Test}

\begin{frame}{Testing}
        \framesubtitle{Why We Test}
        \begin{itemize}
            \item<1-> Making sure that the code behaves as expected
            \item<1-> Preventing \textbf{regressions}
            \item<1-> Easing development and \textbf{maintainability}
            \item<2-> Assessing performance
            \item<2-> Finding and resolving bottlenecks
            \item<2-> Guaranteeing specifications and \textbf{consistency} in usage
        \end{itemize}
\end{frame}

\subsection{How We Test}
\begin{frame}{Testing}
        \framesubtitle{How We Test}
        \begin{itemize}
            \item<1-> Unit and integration tests
                \begin{itemize}
                    \item<1-> Used frequently \textbf{during development}
                    \item<1-> Catches \textbf{regressions}
                \end{itemize}
            \item<2-> System and load testing
                \begin{itemize}
                    \item<2-> To \textbf{gauge performance}
                    \item<2-> Test \textbf{interfacing} with the system
                \end{itemize}
        \end{itemize}
\end{frame}
